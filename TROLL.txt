#include "flipperzero/flipper_zero.h"

void setup() {
  // Várakozás, hogy a Flipper Zero felismerje a számítógép
  delay(3000);

  // Billentyűkombináció a böngésző megnyitásához (Windows: Win + R, majd "chrome")
  pressKey(KEY_GUI);
  pressKey(KEY_R);
  releaseKey(KEY_GUI);
  releaseKey(KEY_R);
  delay(1000);

  // Böngésző nevének beírása
  typeString("chrome");
  delay(500);
  pressKey(KEY_ENTER);
  releaseKey(KEY_ENTER);
  delay(3000);

  // Keresőmezőbe kattintás (feltételezve, hogy a Google a kezdőoldal)
  // Ez a rész rendszerfüggő, lehet, hogy módosítani kell
  // Példa: Tab billentyűvel a keresőmezőre ugrás
  pressKey(KEY_TAB);
  releaseKey(KEY_TAB);
  delay(500);

  // "nyan cat" beírása a keresőbe
  typeString("nyan cat");
  delay(500);
  pressKey(KEY_ENTER);
  releaseKey(KEY_ENTER);
  delay(5000);

  // Első videó kiválasztása (feltételezve, hogy az első találat egy YouTube videó)
  // Ez a rész is rendszerfüggő, lehet, hogy módosítani kell
  pressKey(KEY_TAB);
  releaseKey(KEY_TAB);
    delay(500);
  pressKey(KEY_ENTER);
  releaseKey(KEY_ENTER);

  // Kész!
}
